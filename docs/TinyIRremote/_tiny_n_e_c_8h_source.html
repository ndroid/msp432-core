72// LSB first, 1 start bit + 16 bit address + 8 bit data + 8 bit inverted data + 1 stop bit.
73// see: https://www.sbprojects.net/knowledge/ir/nec.php
74
75#define NEC_ADDRESS_BITS 16 // 16 bit address or 8 bit address and 8 bit inverted address
76#define NEC_COMMAND_BITS 16 // Command and inverted command
77
78#define NEC_BITS (NEC_ADDRESS_BITS + NEC_COMMAND_BITS)
79#define NEC_UNIT 560
80
81#define NEC_HEADER_MARK (16 * NEC_UNIT) // 9000
82#define NEC_HEADER_SPACE (8 * NEC_UNIT) // 4500
83
84#define NEC_BIT_MARK NEC_UNIT
85#define NEC_ONE_SPACE (3 * NEC_UNIT) // 1690
86#define NEC_ZERO_SPACE NEC_UNIT
87
88#define NEC_REPEAT_HEADER_SPACE (4 * NEC_UNIT) // 2250
89
90#define NEC_AVERAGE_DURATION 62000 // NEC_HEADER_MARK + NEC_HEADER_SPACE + 32 * 2,5 * NEC_UNIT + NEC_UNIT // 2.5 because we assume more zeros than ones
91#define NEC_REPEAT_DURATION (NEC_HEADER_MARK + NEC_REPEAT_HEADER_SPACE + NEC_BIT_MARK) // 12 ms
92#define NEC_REPEAT_PERIOD 110000 // Commands are repeated every 110 ms (measured from start to start) for as long as the key on the remote control is held down.
93#define NEC_REPEAT_SPACE (NEC_REPEAT_PERIOD - NEC_AVERAGE_DURATION) // 48 ms
